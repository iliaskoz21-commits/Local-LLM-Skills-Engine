# Use the official Python 3.12 slim image for a lightweight footprint
FROM python:3.12-slim

# Set environment variables to prevent Python from writing .pyc files 
# and to ensure output is sent directly to the terminal (no buffering)
ENV PYTHONDONTWRITEBYTECODE=1
ENV PYTHONUNBUFFERED=1

# Set the working directory inside the container
WORKDIR /app

# Install system dependencies if needed (slim images are very minimal)
RUN apt-get update && apt-get install -y --no-install-recommends \
    build-essential \
    && rm -rf /var/lib/apt/lists/*

# Copy only the requirements file first to leverage Docker's layer caching
COPY requirements.txt .

# Install Python dependencies
RUN pip install --no-cache-dir -r requirements.txt

# Copy the rest of the application code
# This includes app.py, the skills/ directory, and your .env file
COPY . .

# Expose the port (optional, but good for documentation)
# Ollama usually runs on 11434 on the host
EXPOSE 11434

# Command to run the application
# We use -u to ensure the output streams correctly in the Docker console
CMD ["python", "-u", "app.py"]
